弹框使用方法

$(".btn-primary").click(function () {//需要触发事件

var param = {//自定义参数名

title: '提示框', //默认title 提示框

width: "60%",//百分比 数值都可以

height: "400px",//百分比 数值都可以

content: `我们会好好的吗？`,//可以是拼接的标签

innerHTML: true, //false 不支持标签,true 支持标签 默认是false

confirm: function () {},//确认事件

cancel: function () {},//取消事件

};

$.fn.toolTip(param);//初始化方法

});

$.fn.jQuerySelect({

id:'header-select', //id 必传

data:option, //data 可为空

})

tabs使用方法

$.fn.tabs({

id: 'public-tab-1', //id 必传

selected: '下拉框', //设置默认选中项，非必传，默认选中第一个

onSelect: function (val) {, //切换回调事件

console.log(val)

},

})
